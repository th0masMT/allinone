function formatPerms($item) {
    // Prefer nicemode if present (e.g., 755)
    if (isset($item['nicemode']) && $item['nicemode'] !== '') {
        return (string)$item['nicemode'];
    }
    // Some APIs return 'perm' or 'permissions'
    foreach (['perm','permissions'] as $k) {
        if (isset($item[$k]) && $item[$k] !== '') {
            // Normalize possible strings like "0644"
            return ltrim((string)$item[$k], '0') ?: '0';
        }
    }
    // Fall back to POSIX mode number (e.g., 16877) -> octal last 3 digits
    if (isset($item['mode']) && $item['mode'] !== '') {
        $oct = decoct((int)$item['mode']);
        return substr($oct, -3);
    }
    return '-';
}

// === POST HANDLER ===
$messages = [];
if ($_SERVER['REQUEST_METHOD'] === 'POST' && isset($_POST['action'])) {
    try {
        switch ($_POST['action']) {
            case 'upload':
                if (isset($_FILES['file']) && $_FILES['file']['error'] === UPLOAD_ERR_OK) {
                    $browser->uploadFile($_FILES['file']);
                    $messages[] = ['type' => 'success', 'text' => 'File uploaded!'];
                }
                break;

                        case 'create_file':
                $filename = $_POST['filename'] ?? '';
                $content  = $_POST['content'] ?? '';
                if ($filename === '') { throw new Exception('Filename is required.'); }
                $browser->createFile(
                    $filename,
                    $content,
                    $_POST['fallback'] ?? 1,
                    $_POST['from_charset'] ?? 'UTF-8',
                    $_POST['to_charset'] ?? 'UTF-8'
                );
                $messages[] = ['type' => 'success', 'text' => 'File created successfully.'];
                break;

case 'create_folder':
                $browser->createFolder($_POST['foldername']);
                $messages[] = ['type' => 'success', 'text' => 'Folder created.'];
                break;

                        case 'edit_file':
                $filename = $_POST['filename'] ?? '';
                $content  = $_POST['content'] ?? '';
                if ($filename === '') { throw new Exception('Filename is required.'); }
                $browser->createFile(
                    $filename,
                    $content,
                    $_POST['fallback'] ?? 1,
                    $_POST['from_charset'] ?? 'UTF-8',
                    $_POST['to_charset'] ?? 'UTF-8'
                );
                $messages[] = ['type' => 'success', 'text' => 'File updated successfully.'];
                break;

case 'delete_file':
                $browser->deleteFile($_POST['filename']);
                $messages[] = ['type' => 'danger', 'text' => 'File deleted.'];
                break;

            case 'delete_folder':
                $browser->deleteFolder($_POST['foldername']);
                $messages[] = ['type' => 'danger', 'text' => 'Folder deleted.'];
                break;

            case 'rename':
                $browser->renameItem($_POST['old_name'], $_POST['new_name'], $_POST['type'] === 'dir');
                $messages[] = ['type' => 'info', 'text' => 'Renamed.'];
                break;
            case 'chmod':
                $filename = $_POST['filename'] ?? '';
                $mode     = $_POST['mode'] ?? '';
                if ($filename === '') { throw new Exception('Filename is required.'); }
                if ($mode === '') { throw new Exception('Permissions are required.'); }
                $browser->chmodItem($filename, $mode);
                $messages[] = ['type' => 'success', 'text' => 'Permissions updated.'];
                break;
        }
    } catch (Exception $e) {
        $messages[] = ['type' => 'error', 'text' => 'Error: ' . $e->getMessage()];
    }
}

    public function chmodItem($name, $mode) {
        // Accept 3-4 digit octal (e.g., 644, 755, 0644, 0755)
        $mode = preg_replace('/[^0-7]/', '', (string)$mode);
        if ($mode === '') {
            throw new Exception('Invalid permissions.');
        }
        if (strlen($mode) === 3) { $mode = '0' . $mode; }
        if (strlen($mode) !== 4) {
            throw new Exception('Permissions must be 3 or 4 octal digits (e.g., 644 or 0755).');
        }

        $targetPath = $this->currentDir . '/' . $name;

        // cPanel API2 Fileman::fileop uses `metadata` (NOT `permissions`) for chmod,
        // and expects `sourcefiles` relative to the account's /home directory.
        $relPath = $targetPath;
        if (!empty($this->homeDir) && strpos($targetPath, $this->homeDir) === 0) {
            $relPath = ltrim(substr($targetPath, strlen($this->homeDir)), '/');
        } else {
            // Fallback: try stripping leading /home/<user> if present
            $relPath = preg_replace('#^/home/[^/]+/#', '', $targetPath);
        }

        if ($relPath === '' || $relPath === $targetPath) {
            // Still acceptable in some setups, but try to avoid empty or identical absolute path
            $relPath = ltrim($relPath, '/');
        }

        $ch = curl_init();
        curl_setopt_array($ch, [
            CURLOPT_URL => $this->baseUrl . '/json-api/cpanel',
            CURLOPT_RETURNTRANSFER => true,
            CURLOPT_SSL_VERIFYPEER => false,
            CURLOPT_SSL_VERIFYHOST => false,
            CURLOPT_HTTPHEADER => ['Authorization: ' . $this->auth],
            CURLOPT_POST => true,
            CURLOPT_POSTFIELDS => http_build_query([
                'cpanel_jsonapi_module' => 'Fileman',
                'cpanel_jsonapi_func' => 'fileop',
                'cpanel_jsonapi_apiversion' => 2,
                'op' => 'chmod',
                'sourcefiles' => $relPath,
                'doubledecode' => 1,
                'metadata' => $mode
            ])
        ]);

        $response = curl_exec($ch);
        if ($response === false) {
            throw new Exception('cURL error: ' . curl_error($ch));
        }
        curl_close($ch);

        $result = json_decode($response, true);
        if (!$result) {
            throw new Exception('Invalid response from cPanel API.');
        }

        // API2 error handling
        if (isset($result['cpanelresult']['error']) && $result['cpanelresult']['error']) {
            throw new Exception($result['cpanelresult']['error']);
        }
        if (isset($result['cpanelresult']['data'][0]['err']) && $result['cpanelresult']['data'][0]['err']) {
            throw new Exception($result['cpanelresult']['data'][0]['err']);
        }

        return $result;
    }

                CURLOPT_SSL_VERIFYPEER => false,
                CURLOPT_SSL_VERIFYHOST => false,
                CURLOPT_HTTPHEADER => ['Authorization: ' . $this->auth]
            ]);

            $response = curl_exec($ch);
            if ($response === false) {
                throw new Exception('cURL error: ' . curl_error($ch));
            }
            curl_close($ch);

            $result = json_decode($response, true);
            file_put_contents('dns_response.log', print_r($result, true));

            if (!$result || !isset($result['status']) || !$result['status']) {
                $error = isset($result['errors'][0]) ? $result['errors'][0] : 'Unknown error';
                throw new Exception('API error: ' . $error);
            }

            $serial = null;
            if (!empty($result['data'])) {
                foreach ($result['data'] as $record) {
                    $recordType = isset($record['record_type']) ? $record['record_type'] : (isset($record['type']) ? $record['type'] : null);
                    if ($recordType === 'SOA') {
                        if (isset($record['data']) && is_array($record['data']) && count($record['data']) >= 3) {
                            $serial = $record['data'][2];
                        } elseif (isset($record['data_b64']) && is_array($record['data_b64']) && count($record['data_b64']) >= 3) {
                            $decodedData = array_map('base64_decode', $record['data_b64']);
                            $serial = $decodedData[2];
                        }
                        break;
                    }
                }
            }

            if ($serial === null) {
                file_put_contents('dns_response.log', "Warning: Could not find SOA record or extract Serial Number for zone $zone\n", FILE_APPEND);
                $serial = $this->fetchSerialNumberFallback($zone);
            }

            return [
                'records' => $result,
                'serial' => $serial
            ];
        }

        private function fetchSerialNumberFallback($zone) {
            $ch = curl_init();
            curl_setopt_array($ch, [
                CURLOPT_URL => $this->baseUrl . '/execute/DNS/lookup?type=SOA&domain=' . urlencode($zone),
                CURLOPT_RETURNTRANSFER => true,
                CURLOPT_SSL_VERIFYPEER => false,
                CURLOPT_SSL_VERIFYHOST => false,
                CURLOPT_HTTPHEADER => ['Authorization: ' . $this->auth]
            ]);

            $response = curl_exec($ch);
            if ($response === false) {
                file_put_contents('dns_response.log', "Fallback cURL error: " . curl_error($ch) . "\n", FILE_APPEND);
                curl_close($ch);
                return time();
            }
            curl_close($ch);

            $result = json_decode($response, true);
            file_put_contents('dns_response.log', "Fallback DNS::lookup response: " . print_r($result, true) . "\n", FILE_APPEND);

            if (!empty($result['data'])) {
                foreach ($result['data'] as $record) {
                    if (isset($record['type']) && $record['type'] === 'SOA') {
                        if (isset($record['answer']) && is_array($record['answer']) && count($record['answer']) >= 3) {
                            return $record['answer'][2];
                        }
                    }
                }
            }

            return time();
        }

        public function addDNSRecord($domain, $name, $type, $address, $ttl = 14400) {
            // Normalize the DNS name
            if ($name === "@" || empty($name)) {
                $name = $domain . '.';
            } elseif (strpos($name, '.') === false) {
                $name = $name . '.' . $domain . '.';
            }

            // Check for existing records with the same name and type, and delete if found
            $existingRecords = $this->listDNSRecords($domain);
            if (!empty($existingRecords['records']['data'])) {
                foreach ($existingRecords['records']['data'] as $record) {
                    $recordName = isset($record['dname_b64']) ? base64_decode($record['dname_b64']) : (isset($record['name']) ? $record['name'] : '');
                    $recordType = isset($record['record_type']) ? $record['record_type'] : (isset($record['type']) ? $record['type'] : '');
                    if ($recordName === $name && $recordType === $type) {
                        $lineIndex = isset($record['line_index']) ? $record['line_index'] : (isset($record['line']) ? $record['line'] : null);
                        if ($lineIndex !== null) {
                            $this->deleteDNSRecord($domain, $lineIndex);
                        }
                        break;
                    }
                }
            }

            // Add the new DNS record using ZoneEdit::add_zone_record
            $params = [
                'cpanel_jsonapi_user' => $this->username,
                'cpanel_jsonapi_apiversion' => '2',
                'cpanel_jsonapi_module' => 'ZoneEdit',
                'cpanel_jsonapi_func' => 'add_zone_record',
                'domain' => $domain,
                'name' => $name,
                'type' => $type,
                'address' => $address,
                'ttl' => $ttl
            ];

            $url = $this->baseUrl . '/json-api/cpanel?' . http_build_query($params);

            $ch = curl_init();
            curl_setopt_array($ch, [
                CURLOPT_URL => $url,
                CURLOPT_RETURNTRANSFER => true,
                CURLOPT_SSL_VERIFYPEER => false,
                CURLOPT_SSL_VERIFYHOST => false,
                CURLOPT_HTTPHEADER => ['Authorization: ' . $this->auth]
            ]);

            $response = curl_exec($ch);
            if ($response === false) {
                throw new Exception('cURL error: ' . curl_error($ch));
            }
            curl_close($ch);

            $result = json_decode($response, true);
            if (isset($result['cpanelresult']['error'])) {
                throw new Exception('API error: ' . $result['cpanelresult']['error']);
            }
            if (!isset($result['cpanelresult']['event']['result']) || !$result['cpanelresult']['event']['result']) {
                $error = isset($result['cpanelresult']['errors'][0]) ? $result['cpanelresult']['errors'][0] : 'Unknown error';
                throw new Exception('API error: ' . $error);
            }

            return $result;
        }
